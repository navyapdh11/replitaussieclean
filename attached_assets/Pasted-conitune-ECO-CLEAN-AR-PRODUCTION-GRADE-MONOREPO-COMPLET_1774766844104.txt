## 🔐 `apps/api/src/auth/auth.guard.ts` (Production Security)

```typescript
import {
  CanActivate,
  ExecutionContext,
  Injectable,
  UnauthorizedException,
  ForbiddenException,
} from '@nestjs/common';
import { Reflector } from '@nestjs/core';
import { JwtService } from '@nestjs/jwt';
import * as argon2 from 'argon2';

@Injectable()
export class AuthGuard implements CanActivate {
  constructor(
    private jwtService: JwtService,
    private reflector: Reflector,
  ) {}

  async canActivate(context: ExecutionContext): Promise<boolean> {
    const isPublic = this.reflector.getAllAndOverride<boolean>('isPublic', [
      context.getHandler(),
      context.getClass(),
    ]);
    if (isPublic) {
      return true;
    }

    const request = context.switchToHttp().getRequest();
    const token = this.extractTokenFromHeader(request);
    if (!token) {
      throw new UnauthorizedException();
    }
    try {
      const payload = await this.jwtService.verifyAsync(token);
      request['user'] = payload;
    } catch {
      throw new UnauthorizedException();
    }
    return true;
  }

  private extractTokenFromHeader(request: Request): string | undefined {
    const [type, token] = request.headers['authorization']?.split(' ') ?? [];
    return type === 'Bearer' ? token : undefined;
  }
}

@Injectable()
export class RolesGuard implements CanActivate {
  constructor(private reflector: Reflector) {}

  canActivate(context: ExecutionContext): boolean {
    const requiredRoles = this.reflector.getAllAndOverride<Role[]>(
      'roles',
      [context.getHandler(), context.getClass()],
    );
    if (!requiredRoles) {
      return true;
    }
    const { user } = context.switchToHttp().getRequest();
    return requiredRoles.some((role) => user.role?.includes(role));
  }
}
```

```typescript
import { StateGraph, END, START } from '@langgraph/core';
import { ChatOpenAI } from '@langchain/openai';
import { Redis } from '@upstash/redis';
import { z } from 'zod';
import { RateLimiter } from 'limiter';
import { v4 as uuidv4 } from 'uuid';
import pino from 'pino';

// === PRODUCTION CONFIGURATION ===
const config = {
  openai: {
    model: 'gpt-4o-2026-01-01',
    temperature: 0.1,
    maxTokens: 2000,
    fallbackModels: ['gpt-4o-mini-2024-07-18'],
  },
  redis: {
    url: process.env.REDIS_URL!,
  },
  rateLimits: {
    llmCalls: { windowMs: 60000, max: 30 },
  },
  agentTimeouts: {
    maxMs: 30000,
  },
};

// === PRODUCTION LOGGING ===
const logger = pino({
  level: process.env.LOG_LEVEL || 'info',
  redact: ['OPENAI_API_KEY', 'STRIPE_SECRET_KEY'],
  transport: {
    target: 'pino-pretty',
    options: {
      colorize: true,
    },
  },
});

// === ENHANCED INPUT VALIDATION ===
const BaseAgentInput = z.object({
  traceId: z.string().optional().default(() => uuidv4()),
  timestamp: z.date().default(() => new Date()),
  customerId: z.string().optional(),
  sessionId: z.string().optional(),
});

const SchedulerInput = BaseAgentInput.extend({
  bookingId: z.string(),
  serviceId: z.string().regex(/^ECO-\d{3}$/),
  customerLocation: z.object({
    suburb: z.string().max(50),
    postcode: z.string().regex(/^\d{4}$/),
    lat: z.number().min(-90).max(90),
    lng: z.number().min(-180).max(180),
  }),
  preferredTime: z.string().datetime(),
  durationMinutes: z.number().int().min(30).max(480),
  priority: z.enum(['STANDARD', 'HIGH', 'URGENT']).default('STANDARD'),
});

const PricingInput = BaseAgentInput.extend({
  serviceId: z.string().regex(/^ECO-\d{3}$/),
  basePrice: z.number().positive().max(10000),
  demandScore: z.number().int().min(0).max(10),
  capacityUtilization: z.number().min(0).max(1),
  timeFactors: z.array(z.enum(['PEAK', 'WEEKEND', 'HOLIDAY'])).max(5),
  customerSegment: z.enum(['RESIDENTIAL', 'COMMERCIAL', 'ENTERPRISE']),
  bookingLeadTimeHours: z.number().int().min(0).max(168),
});

// === PRODUCTION RAG ENGINE ===
class ProductionRAGEngine {
  private redis: Redis;
  private llm: ChatOpenAI;
  private limiter: RateLimiter;

  constructor() {
    this.redis = new Redis({ url: config.redis.url });
    this.llm = new ChatOpenAI({
      model: config.openai.model,
      temperature: config.openai.temperature,
      maxTokens: config.openai.maxTokens,
    });
    this.limiter = new RateLimiter({
      tokensPerInterval: config.rateLimits.llmCalls.max,
      interval: config.rateLimits.llmCalls.windowMs,
    });
  }

  async search(
    query: string,
    filters: Record<string, any> = {},
    topK: number = 5,
  ): Promise<any[]> {
    const traceId = uuidv4();
    
    try {
      // Rate limiting
      await this.limiter.removeTokens(1);
      
      // Sanitize query (prevent prompt injection)
      const sanitizedQuery = this.sanitizeQuery(query);
      
      // Vector search (Redis HNSW)
      const results = await this.redis.ft().search({
        index: 'rag_index',
        query: `@content:(${sanitizedQuery})`,
        filter: filters,
        limit: topK,
        returnFields: ['content', 'metadata', 'score'],
      }) as any;

      logger.info('RAG search completed', { traceId, query: sanitizedQuery, hitCount: results.total });

      return results.documents.map((doc: any) => ({
        content: doc.content,
        metadata: JSON.parse(doc.metadata),
        score: parseFloat(doc.score),
      }));
    } catch (error) {
      logger.error('RAG search failed', { traceId, error });
      return [];
    }
  }

  private sanitizeQuery(query: string): string {
    // Remove potential injection patterns
    const injectionPatterns = [
      'ignore previous',
      'forget everything',
      'override instructions',
      'system prompt',
    ];
    return injectionPatterns.reduce(
      (clean, pattern) => clean.replace(new RegExp(pattern, 'gi'), '[REDACTED]'),
      query,
    );
  }
}

// === ENHANCED MULTI-AGENT ORCHESTRATOR ===
class ProductionMultiAgentSystem {
  private graph: StateGraph;
  private rag: ProductionRAGEngine;
  private redis: Redis;
  private fallbackLLM: ChatOpenAI;

  constructor() {
    this.redis = new Redis({ url: config.redis.url });
    this.rag = new ProductionRAGEngine();
    this.fallbackLLM = new ChatOpenAI({
      model: config.openai.fallbackModels[0],
      temperature: 0.1,
    });
    
    this.graph = this.buildProductionGraph();
  }

  private buildProductionGraph() {
    const workflow = new StateGraph({
      channels: {
        input: null,
        intent: null,
        schedulerResult: null,
        pricingResult: null,
        supportResult: null,
        error: null,
        traceId: null,
      },
    });

    // Router Node with fallback
    workflow.addNode('router', this.routerNode.bind(this));
    workflow.addNode('scheduler', this.schedulerNode.bind(this));
    workflow.addNode('pricing', this.pricingNode.bind(this));
    workflow.addNode('support', this.supportNode.bind(this));
    workflow.addNode('error_handler', this.errorHandlerNode.bind(this));

    // Edges with error handling
    workflow.addEdge(START, 'router');
    workflow.addConditionalEdges('router', this.routeToAgent);
    workflow.addEdge('scheduler', END);
    workflow.addEdge('pricing', END);
    workflow.addEdge('support', END);
    workflow.addEdge('error_handler', END);

    return workflow.compile();
  }

  private async routerNode(state: any) {
    try {
      const input = BaseAgentInput.parse(state.input);
      
      const prompt = `
Classify intent from user input: "${state.input?.query || state.input}"

Options: booking, pricing, support, marketing, seo
Output ONLY the intent name.
`;

      const result = await this.safeLLMCall(prompt);
      return {
        ...state,
        intent: result.intent || 'support',
        traceId: input.traceId,
      };
    } catch (error) {
      logger.error('Router failed', { error });
      return { ...state, intent: 'support', error: error.message };
    }
  }

  private async schedulerNode(state: any) {
    const input = SchedulerInput.parse(state.input);
    const context = await this.rag.search(
      `scheduling ${input.serviceId} ${input.customerLocation.suburb}`,
    );

    const prompt = `
You are a production scheduling optimizer for EcoClean AR.

INPUT: ${JSON.stringify(input, null, 2)}
RAG Context: ${context.slice(0, 3).map(c => c.content).join('\n\n')}

Available Cleaners Today:
${JSON.stringify(state.availableCleaners || [], null, 2)}

Output validated JSON:
${SchedulerOutput.schema()}
`;

    const result = await this.safeLLMCall(prompt);
    return {
      ...state,
      schedulerResult: result,
      assignedCleaner: result.assignedCleaner,
    };
  }

  private async safeLLMCall(prompt: string, maxRetries = 2): Promise<any> {
    for (let attempt = 1; attempt <= maxRetries; attempt++) {
      try {
        const response = await this.rag.llm.invoke(prompt);
        const parsed = z.object({}).passthrough().parse(JSON.parse(response.content as string));
        return parsed;
      } catch (error) {
        logger.warn(`LLM call failed (attempt ${attempt})`, { error });
        
        if (attempt === maxRetries) {
          // Fallback to cheaper model
          const fallbackResponse = await this.rag.fallbackLLM.invoke(prompt);
          return JSON.parse(fallbackResponse.content as string);
        }
      }
    }
    throw new Error('All LLM calls failed');
  }

  async run(input: any) {
    const traceId = uuidv4();
    const startTime = Date.now();

    try {
      const result = await this.graph.invoke({ input, traceId });
      
      logger.info('Agent orchestration completed', {
        traceId,
        durationMs: Date.now() - startTime,
        intent: result.intent,
      });

      return result;
    } catch (error) {
      logger.error('Orchestration failed', { traceId, error });
      throw error;
    }
  }
}

const mas = new ProductionMultiAgentSystem();
export { mas as orchestrator };
```

```tsx
    if ('xr' in navigator) {
      (navigator as any).xr.isSessionSupported('immersive-ar').then((supported: boolean) => {
        setARSupported(supported);
      });
    }
  }, []);

  const handleScanComplete = (roomData: any) => {
    onScanComplete(roomData);

        <ambientLight intensity={0.5} />
        <ARButton

          onSessionStart={(session) => setSession(session)}
        />
        
        <RoomScanner onScanComplete={handleScanComplete} />
        <OrbitControls />

    </div>

}
```

## 🛡️ `apps/api/src/common/guards/rate-limit.guard.ts`

```typescript
import {
  ExecutionContext,
  Injectable,
  Inject,
  CACHE_MANAGER,
  ForbiddenException,
} from '@nestjs/common';
import { Cache } from 'cache-manager';
import { InjectRedis } from '@liaoliaots/nestjs-redis';
import { Redis } from 'ioredis';

@Injectable()
export class RateLimitGuard {
  constructor(
    @InjectRedis() private readonly redis: Redis,
    @Inject(CACHE_MANAGER) private cacheManager: Cache,
  ) {}

  async canActivate(context: ExecutionContext): Promise<boolean> {
    const request = context.switchToHttp().getRequest();
    const clientIp = this.getClientIp(request);
    const endpointKey = request.route.path;
    
    const key = `rate_limit:${clientIp}:${endpointKey}`;
    const windowMs = 15 * 60 * 1000; // 15 minutes
    const maxRequests = 100;

    const [requests] = await this.redis
      .multi()
      .incr(key)
      .expire(key, windowMs / 1000)
      .exec();

    if ((requests[1] as number) > maxRequests) {
      throw new ForbiddenException('Rate limit exceeded');
    }

    return true;
  }

  private getClientIp(request: any): string {
    return (
      request.headers['x-forwarded-for']?.split(',')[0]?.trim() ||
      request.ip ||
      request.connection.remoteAddress ||
      'unknown'
    );
  }
}
```
