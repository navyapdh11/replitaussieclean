### 🆕 FEATURE 3: CLEANER FLEET MANAGER (Real-time Dashboard)

`apps/web/app/admin/fleet/page.tsx`
```tsx
'use client';

import { useEffect, useState } from 'react';
import { MapContainer, TileLayer, Marker, Popup } from 'react-leaflet';
import { api } from '@/trpc/react';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';

export default function FleetManager() {
  const { data: cleaners, refetch } = api.cleaners.realTime.useQuery();
  const [selectedCleaner, setSelectedCleaner] = useState<any>(null);

  useEffect(() => {
    const interval = setInterval(refetch, 30000); // 30s refresh
    return () => clearInterval(interval);
  }, [refetch]);

  if (!cleaners) return <div>Loading fleet...</div>;

  return (
    <div className="space-y-6">
      {/* Fleet Summary Cards */}
      <div className="grid grid-cols-1 md:grid-cols-4 gap-4">
        <Card>
          <CardHeader>
            <CardTitle>Active Cleaners</CardTitle>
          </CardHeader>
          <CardContent>
            <div className="text-4xl font-bold">
              {cleaners.filter(c => c.status === 'ACTIVE').length}
            </div>
          </CardContent>
        </Card>
        {/* Utilization, Avg Rating, Revenue Today */}
      </div>

      {/* Real-time Map */}
      <Card>
        <CardHeader>
          <CardTitle>Real-time Cleaner Tracking</CardTitle>
        </CardHeader>
        <CardContent className="h-[500px]">
          <MapContainer center={[-32.24, 115.77]} zoom={11} style={{ height: '100%' }}>
            <TileLayer url="https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png" />
            {cleaners.map((cleaner) => (
              <Marker 
                key={cleaner.id}
                position={[cleaner.currentLat, cleaner.currentLng]}
                eventHandlers={{
                  click: () => setSelectedCleaner(cleaner),
                }}
              >
                <Popup>
                  <div>
                    <h3 className="font-bold">{cleaner.name}</h3>
                    <p>Status: {cleaner.status}</p>
                    <p>Current Job: {cleaner.currentJob || 'Available'}</p>
                    <p>ETA: {cleaner.etaMinutes} min</p>
                  </div>
                </Popup>
              </Marker>
            ))}
          </MapContainer>
        </CardContent>
      </Card>

      {/* Cleaner Details Modal */}
      {selectedCleaner && (
        <CleanerDetailsModal cleaner={selectedCleaner} onClose={() => setSelectedCleaner(null)} />
      )}
    </div>
  );
}
```

```typescript
export class DynamicPricingEngine {
  private factors: PricingFactor[] = [
    { name: 'PEAK_HOUR', multiplier: 1.25, condition: (time: Date) => time.getHours() >= 15 && time.getHours() <= 18 },
    { name: 'WEEKEND', multiplier: 1.15, condition: (time: Date) => time.getDay() === 0 || time.getDay() === 6 },
    { name: 'HIGH_DEMAND', multiplier: 1.3, condition: (utilization: number) => utilization > 0.8 },
    { name: 'LTV_DISCOUNT_HIGH', multiplier: 0.9, condition: (ltvTier: string) => ltvTier === 'HIGH' },
    { name: 'SUBSCRIPTION', multiplier: 0.85, condition: (isSub: boolean) => isSub },
    { name: 'REFERRAL', multiplier: 0.95, condition: (referralCount: number) => referralCount > 0 },
  ];

  calculateOptimalPrice(basePrice: number, context: PricingContext): number {
    let finalPrice = basePrice;
    const appliedFactors: AppliedFactor[] = [];

    for (const factor of this.factors) {
      if (factor.condition(context)) {
        finalPrice *= factor.multiplier;
        appliedFactors.push({ factor: factor.name, multiplier: factor.multiplier });
      }
    }

    // Margin guard (minimum 60%)
    const margin = 1 - (basePrice * 0.15) / finalPrice;
    if (margin < 0.6) {
      finalPrice = Math.ceil((basePrice * 0.15) / 0.6 * 100) / 100;
    }

    return Math.round(finalPrice * 20) / 20; // Nearest $0.05
  }
}

interface PricingContext {
  time: Date;
  utilization: number;
  ltvTier: string;
  isSubscription: boolean;
  referralCount: number;
  demandScore: number;
}
```
